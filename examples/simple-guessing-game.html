#include <iomanip>
#include <iostream>
#include <unistd.h>

using namespace std;

bool isNumber(const string &s) {
  for (char const &ch : s) {
    if (isdigit(ch) == 0)
      return false;
  }
  return true;
}

int main() {
  int points = 5;
  while (true) {
    int guess = 0;
    string tempGuess;
    cout << endl << "Enter a guess: " << endl;
    getline(cin, tempGuess);
    if (!isNumber(tempGuess)) {
      return 0;
    }
    tempGuess = tempGuess.substr(0,4);
    guess = stoi(tempGuess);
    system("clear");
    int time_before_loop_begins = time(NULL);
    int realNumber = 0;
    while (time(NULL) - time_before_loop_begins < 3) {
      system("clear");
      int outputNumber = rand() % 10001;
      cout << "Current points: " << points << endl << endl;
      cout << setfill('0') << setw(4) << guess << endl << flush;
      cout << setfill('0') << setw(4) << outputNumber << endl << flush;
      realNumber = outputNumber;
      usleep(0100000);
    }
    if (guess == realNumber) {
      cout << "You got it!!! +6 points \n";
      points += 6;
    } else if (abs(guess - realNumber) < 500) {
      cout << "So close!!! +3 points \n";
      points += 3;
    } else if (abs(guess - realNumber) < 1500) {
      cout << "Pretty good!!! +1 point \n";
      points += 1;
    } else if (abs(guess - realNumber) > 3000) {
      cout << "You are so off!!! -1 point \n";
      points -= 1;
    }
  }
}
  • First, you have to enter a number that consists of four digits. For example, you can enter or , but not or . If you enter a number that is lower than four digits, it will add the optional zeros. If you enter a number that is greater than four digits, it will truncate the extra digits.
  • Second, the program will display a four-digit number multiple times per second on the screen, along with your number above it. The four-digit numbers are randomly generated by the program and may or may not be close to your number.
  • Third, after a certain amount of time has elapsed (in this case, one second), the program will evaluate your guess and assign you points. The points are based on how close your number is to the actual number that the program has chosen. The closer you are, the more points you get. The farther you are, the fewer points you get. If you guess the exact number, you get the maximum points possible.

Core concepts
...

  • getline(cin, tempGuess); is used in order to take input from the user
  • isNumber( &s ) passes a reference to a string into the method
  • isdigit(ch) checks if a inputted char is a digit
  • .substr(0,4) ensures the string is a maximum of 4 digits
  • stoi() converts the input string to an integer that can be worked with
  • system("clear"); clears the console screen
  • time(NULL); is a time object representing the current time
  • rand() generates a random number in the range of [0, RAND_MAX)
  • << setfill('0') << setw(4) setfill determines what empty integer spaces should be and setw determines the width of the integers
  • usleep(0100000); will pause the program for the set number of microseconds
  • abs() which gets the absolute value of a number
