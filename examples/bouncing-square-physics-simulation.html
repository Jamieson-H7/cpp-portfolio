#include <chrono>
#include <iostream>
#include <stdio.h>
#include <sys/ioctl.h>
#include <thread>
#include <unistd.h>

using namespace std;

class Square {
private:
  int width;
  int height;

public:
  double x = 0;
  double y = 0;
  Square(int inpWid, int inpHei) {
    width = inpWid;
    height = inpHei > 3 ? inpHei * 0.55 : inpHei;
  }
  string toString() {
    string endString = "";
    for (int j = 0; j < height; j++) {
      if (j == 0) {
        for (int i = 0; i < width; i++) {
          endString += '_';
        }
      } else {
        if (width > 1) {
          endString += "⎸";
          for (int i = 0; i < width - 2; i++) {
            endString += ' ';
          }
          endString += "⎹";
        } else {
          for (int i = 0; i < width - 2; i++) {
            endString += '0';
          }
        }
      }
      endString += '\n';
      for (int k = 0; k < x; k++) {
        endString += ' ';
      }
    }
    for (int i = 0; i < width; i++) {
      endString += "¯";
    }

    return endString;
  }
  int getWidth() { return width; }
  int getHeight() { return height; }
};

int main() {
  Square player = Square(10, 10);
  player.y = 10;
  // int x = 0;
  // int y = 0;
  int max_x = 0;
  int max_y = 0;
  int next_x = 0;
  int next_y = 0;
  double directionX = 5;
  double directionY = -0.5;
  double gravity = 0.05;
  // Get the width of the terminal

  while (1) {
    struct winsize w;
    ioctl(STDOUT_FILENO, TIOCGWINSZ, &w);
    max_x = w.ws_col;
    max_y = w.ws_row - 2;
    //  Clear the screen
    cout << "\033[2J\033[1;1H";

    // Print spaces before the "ball"
    for (int i = 0; i < player.y; i++) {
      cout << '\n';
    }
    for (int i = 0; i < player.x; i++) {
      cout << ' ';
    }

    cout << player.toString(); // Print our "ball"
    cout << "\n";

    this_thread::sleep_for(
        chrono::milliseconds(40)); // Shorter delay between movements

    next_x = player.x + directionX;
    next_y = player.y + directionY;

    if (next_x >= max_x - player.getWidth() || next_x < 0) {
      directionX *= -1;
    } else {
      player.x += directionX;
    }
    if (next_y >= max_y - player.getHeight() || next_y < 0) {
      directionX *= 0.95;
      directionY -= 0.1;
      directionY *= -1;
    } else {
      player.y += directionY;
    }
    directionY += gravity;
  }
  while (player.y + player.getHeight() > max_y) {
    player.y -= 1;
  }
  while (player.x + player.getWidth() > max_x) {
    player.x -= 1;
  }
  if (player.x < 0) {
    player.x = 1;
  }

  return 0;
}

  • #include lines. These lines are including various C++ standard library headers necessary for input/output operations (iostream), time measurement (chrono), and terminal control (ioctl, unistd.h). Also included is the C standard I/O library (stdio.h) and the C++ threading library (thread) for working with threads.
  • class Square This code defines a class called Square which will be used to create the object for the physics simulation
  • private: The square class contains private data members of width and height. This means they can not be changed from outside of the class directly and require code inside the class to modify them.
  • public: The square class contains public data members of x and y, and a constructor for Square. This allows the x and y data to be modified outside the class and allows the class constructor to be called.
  • Square(int inpWid, int inpHei) initializes the width and height based on the input parameters, and x and y are initialized to 0. This constructor is a public constructor which allows it to be created outside of the class.
  • toString() isn't defined by default in C++. This created version returns a string representation of a square with _, ⎸, and ¯ characters.
  • getWidth() returns the width of the square, and getHeight() returns the height of the square.
  • Square player = Square(10, 10); an instance of the Square class named player is created with a width and height of 10.
  • struct winsize w; The winsize structure is used to hold the dimensions of a terminal window.
  • ioctl(STDOUT_FILENO, TIOCGWINSZ, &w); This is used to get the size of the terminal window. ioctl is a system call with parameters STDOUT_FILENO which is a file descriptor that refers to the terminal and  TIOCGWINSZ which is a request code that asks ioctl to get the window size. &w is a pointer to the winsize structure for ioctl to store the dimensions of the terminal window to.
  • max_x = w.ws_col; and max_y = w.ws_row - 2; Set the max x and max y positions based on the console size.
  • cout << "\033[2J\033[1;1H"; This moves the cursor to the top-left corner of the terminal and clears the entire screen using ANSI escape codes.
  • this_thread::sleep_for( chrono::milliseconds(40)); This requires the program to sleep for 40 milliseconds for each frame of the animation.
